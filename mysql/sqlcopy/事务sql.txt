START TRANSACTION;# 开启事务
BEGIN;# 开启事务
SELECT @@autocommit;# 查看事务提交方式
SET @@autocommit=0;# 设置事务提交方式-手动
SET @@autocommit=1;# 设置事务提交方式-自动
COMMIT;# 提交事务
ROLLBACK;# 回滚事务

SELECT @@TRANSACTION_ISOLATION;# 查看事务隔离级别
SET [ SESSION | GLOBAL ] TRANSACTION ISOLATION LEVEL{READ UNCOMMITTED | READ COMMITTED | REPEATABLE READ | SERIALIZABLE};# 设置事务隔离级别

# 当前会话
SET SESSION TRANSACTION ISOLATION LEVEL READ UNCOMMITTED;# 读未提交*引发脏读、不可重复读、幻读
SET SESSION TRANSACTION ISOLATION LEVEL READ COMMITTED;# 读已提交*引发不可重复读、幻读
SET SESSION TRANSACTION ISOLATION LEVEL REPEATABLE READ;# 可重复读*引发幻读
SET SESSION TRANSACTION ISOLATION LEVEL SERIALIZABLE;# 可串行化*无

# 全局设置
SET GLOBAL TRANSACTION ISOLATION LEVEL READ UNCOMMITTED;  -- 全局读未提交（风险高）
SET GLOBAL TRANSACTION ISOLATION LEVEL READ COMMITTED;    -- 全局读已提交（如 Oracle 默认）
SET GLOBAL TRANSACTION ISOLATION LEVEL REPEATABLE READ;   -- 全局可重复读（如 MySQL 默认）
SET GLOBAL TRANSACTION ISOLATION LEVEL SERIALIZABLE;      -- 全局可串行化（性能低）
